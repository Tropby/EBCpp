90  virtual bool open( EBIODevice::DIRECTION direction )
91  {
92 
93  if( direction != READ_WRITE )
94  EB_EXCEPTION( "Can not open a tcp socket read only or write only." );
95 
96  // Check if hostname can be used for this connection
97  std::string host = getFileName();
98 
99  if( host.substr(0, 6).compare("tcp://") )
100  EB_EXCEPTION( "EBTcpSocket needs a tcp://{hostname}:{port} filename to connect to a host." );
101 
102  host = host.substr(6);
103 
104  if( host.find(':') == std::string::npos )
105  EB_EXCEPTION( "EBTcpSocket needs a tcp://{hostname}:{port} filename to connect to a host." );
106 
107  int32_t port = std::stoi( host.substr( host.find(':') + 1 ) );
108  host = host.substr( 0, host.find(':'));
109 
110  if( port > 0xFFFF || port < 0 )
111  EB_EXCEPTION( "EBTcpSocket port rage (0 - 65535)." );
112 
113  // Get ip address from host name otherwise treat host as ip address
114  std::string hostIp = EBUtils::hostnameToIp(host);
115  if (!hostIp.empty())
116  host = hostIp;
117 
118  // convert ip address string to ip address
119  #ifdef __WIN32__
120  memset(&address,0,sizeof(SOCKADDR_IN));
121  address.sin_family=AF_INET;
122  address.sin_port=htons(port);
123  address.sin_addr.s_addr=inet_addr(host.c_str());
124  #else
125  inet_pton(AF_INET, host.c_str(), &address);
126  #endif
127 
128  startThread();
129 
130  return true;
131  }

306  void run()
307  {
308  EBUtils::setThreadName( "TcpSocket #???" );
309 
310  // Socket is allredy known and connected (eg. server sockets)
311  if( socketId == -1 )
312  {
313  // Try to get a new socket
314  socketId = ::socket( AF_INET, SOCK_STREAM, 0);
315  if( socketId == -1 )
316  {
317  EB_EMIT_WITH_ARGS( error, "Can not create socket!" );
318  return;
319  }
320 
321  // Try to connect to host
322  int descriptor = ::connect(socketId, reinterpret_cast<sockaddr*>(&address), sizeof(address));
323  if (descriptor == -1)
324  {
325  EB_EMIT_WITH_ARGS( error, "Can not connect to address!" );
326  return;
327  }
328  connectionState = true;
329  EB_EMIT(connected);
330  }
331  else
332  {
333  connectionState = true;
334  }
335 
336  EBUtils::setThreadName( std::string("TcpSocket #") + std::to_string(socketId) );
337 
338  // Read until the end of life
339  char buffer[1024];
340  int nbytes;
341  while( true )
342  {
343  // Read next block of data
344  nbytes = ::recv(socketId, buffer, sizeof(buffer), 0);
345 
346  switch (nbytes)
347  {
348  // Discriptor disconnected
349  case 0:
350  EB_EMIT(disconnected);
351  return;
352 
353  case -1:
354  // if not closed send error message otherwise send disconnected
355  if( connectionState )
356  {
357  EB_EMIT_WITH_ARGS(error, "Error while reading from socket!");
358  }
359  else
360  {
361  EB_EMIT(disconnected);
362  }
363  return;
364 
365  default:
366  std::vector<uint8_t> b(buffer, buffer + nbytes);
367  data.insert(data.end(), std::begin(b), std::end(b));
368  EB_EMIT(readReady);
369  }
370  }
371 
372  return;
373  }
