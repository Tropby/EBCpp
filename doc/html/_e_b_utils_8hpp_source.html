52  static std::string hostnameToIp(std::string hostname)
53  {
54  struct addrinfo hints, *servinfo, *p;
55  struct sockaddr_in *h;
56  int rv;
57 
58  memset(&hints, 0, sizeof hints);
59  hints.ai_family = AF_INET;
60  hints.ai_socktype = SOCK_STREAM;
61 
62  if ((rv = getaddrinfo(hostname.c_str(), "http", &hints, &servinfo)) != 0)
63  {
64  return "";
65  }
66 
67  // loop through all the results and connect to the first we can
68  char ip[64];
69  for (p = servinfo; p != NULL; p = p->ai_next)
70  {
71  h = (struct sockaddr_in*) p->ai_addr;
72  strcpy(ip, inet_ntoa(h->sin_addr));
73  }
74 
75  freeaddrinfo(servinfo); // all done with this structure
76  return ip;
77 
78  }

113  static std::string trim(const std::string &s)
114  {
115  auto isWhitespace = [](const char c)
116  {
117  return c==' ' || c=='\n' || c=='\t' || c=='\r' || c==11;
118  };
119 
120  int left=0, right=s.length()-1;
121 
122  while(left<=right && isWhitespace(s[left]))
123  ++left;
124 
125  while(right>left && isWhitespace(s[right]))
126  --right;
127 
128  return s.substr(left, 1+right-left);
129  }
130 };
