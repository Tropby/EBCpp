91  virtual bool open( EBIODevice::DIRECTION direction )
92  {
93 
94  if( direction != READ_WRITE )
95  EB_EXCEPTION( "Can not open a tcp socket read only or write only." );
96 
97  // Check if hostname can be used for this connection
98  std::string host = getFileName();
99 
100  if( host.substr(0, 6).compare("tcp://") )
101  EB_EXCEPTION( "EBTcpSocket needs a tcp://{hostname}:{port} filename to connect to a host." );
102 
103  host = host.substr(6);
104 
105  if( host.find(':') == std::string::npos )
106  EB_EXCEPTION( "EBTcpSocket needs a tcp://{hostname}:{port} filename to connect to a host." );
107 
108  int32_t port = std::stoi( host.substr( host.find(':') + 1 ) );
109  host = host.substr( 0, host.find(':'));
110 
111  if( port > 0xFFFF || port < 0 )
112  EB_EXCEPTION( "EBTcpSocket port rage (0 - 65535)." );
113 
114  // Get ip address from host name otherwise treat host as ip address
115  std::string hostIp = EBUtils::hostnameToIp(host);
116  if (!hostIp.empty())
117  host = hostIp;
118 
119  // convert ip address string to ip address
120  #ifdef __WIN32__
121  memset(&address,0,sizeof(SOCKADDR_IN));
122  address.sin_family=AF_INET;
123  address.sin_port=htons(port);
124  address.sin_addr.s_addr=inet_addr(host.c_str());
125  #else
126  inet_pton(AF_INET, host.c_str(), &address);
127  #endif
128 
129  startThread();
130 
131  return true;
132  }

309  virtual bool connect()
310  {
311  // Try to get a new socket
312  socketId = ::socket( AF_INET, SOCK_STREAM, 0);
313  if( socketId == -1 )
314  {
315  EB_EMIT_WITH_ARGS( error, "Can not create socket!" );
316  return false;
317  }
318 
319  // Try to connect to host
320  int descriptor = ::connect(socketId, reinterpret_cast<sockaddr*>(&address), sizeof(address));
321  if (descriptor == -1)
322  {
323  EB_EMIT_WITH_ARGS( error, "Can not connect to address!" );
324  return false;
325  }
326  connectionState = true;
327  return true;

353  void run()
354  {
355  EBUtils::setThreadName( "TcpSocket #???" );
356 
357  // Socket is allredy known and connected (eg. server sockets)
358  if( socketId == -1 )
359  {
360  if( !connect() )
361  {
362  return;
363  }
364  connected.emit(this);
365 
366  //EB_EMIT(connected);
367  }
368  else
369  {
370  connectionState = true;
371  }
372 
373  EBUtils::setThreadName( std::string("TcpSocket #") + std::to_string(socketId) );
374 
375  // Read until the end of life
376  char buffer[1024];
377  int nbytes;
378  while( true )
379  {
380  nbytes = receiveData( buffer, sizeof(buffer) );
381 
382  switch (nbytes)
383  {
384  // Discriptor disconnected
385  case 0:
386  EB_EMIT(disconnected);
387  return;
388 
389  case -1:
390  // if not closed send error message otherwise send disconnected
391  if( connectionState )
392  {
393  EB_EMIT_WITH_ARGS(error, "Error while reading from socket!");
394  }
395  else
396  {
397  EB_EMIT(disconnected);
398  }
399  return;
400 
401  default:
402  std::vector<uint8_t> b(buffer, buffer + nbytes);
403  mutex.lock();
404  data.insert(data.end(), std::begin(b), std::end(b));
405  mutex.unlock();
406  EB_EMIT(readReady);
407  }
408  }
409 
410  return;
411  }
